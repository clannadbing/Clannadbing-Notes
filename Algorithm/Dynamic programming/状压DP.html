解题思路

仿照 排列型回溯 的定义方式，我们需要知道当前还有哪些数（下标）可以选，以及上一个选的数（下标）是多少。

定义 dfs(i,j) 表示当前可以选的下标集合为 i，上一个选的数的下标是 j 时，可以构造出多少个特别排列。

枚举当前要选的数的下标 k，如果 nums[k] 与 nums[j] 满足题目整除的要求，则

dfs(i,j)=∑k∈idfs(i∖{k},k)

递归边界：dfs(0,j)=1，表示找到了一个特别排列。

递归入口：dfs(U∖{j},j)，其中全集 U={0,1,2,...,n−1}。枚举特别排列的第一个数的下标 j，累加所有 dfs(U∖{j},j)，即为答案。

记忆化搜索

// 函数在外面
const int MOD = 1e9 + 7;
class Solution {
public:
    int dfs(int s, int i, int n, vector<int>& nums, vector<vector<int>>& memo) {
        if (s == 0) {
            return 1;
        }
        if (memo[s][i] != -1) {
            return memo[s][i];
        }
        int res = 0;
        for (int j = 0; j < n; j++) {
            if (((s >> j) & 1) == 1 && (nums[i] % nums[j] == 0 || nums[j] % nums[i] == 0)) {
                res = (res + dfs(s ^ (1 << j), j, n, nums, memo)) % MOD;
            }
        }
        memo[s][i] = res;
        return res;
    }
    int specialPerm(vector<int>& nums) {
        int n = nums.size();
        int s = (1 << n) - 1;
        int ans = 0;
        vector<vector<int>> memo(1<<n, vector<int>(n, -1));
        for (int i = 0; i < n; i++) {
            ans = (ans + dfs(s ^ (1 << i), i, n, nums, memo)) % MOD; 
        }
        return ans;
    }
};

const int MOD = 1e9 + 7;
class Solution {
public:
    int specialPerm(vector<int>& nums) {
        int n = nums.size();
        int s = (1 << n) - 1;
        int ans = 0;
        vector<vector<int>> memo(1 << n, vector<int>(n, -1));
        function<int(int, int)> dfs = [&](int s, int i) -> int {
            if (s == 0) {
                return 1;
            }
            if (memo[s][i] != -1) {
                return memo[s][i];
            }
            int res = 0;
            for (int j = 0; j < n; j++) {
                if (((s >> j) & 1) == 1 && (nums[i] % nums[j] == 0 || nums[j] % nums[i] == 0)) {
                    res = (res + dfs(s ^ (1 << j), j)) % MOD;
                }
            }
            memo[s][i] = res;
            return res;
        };
        for (int i = 0; i < n; i++) {
            ans = (ans + dfs(s ^ (1 << i), i)) % MOD; 
        }
        return ans;
    }
};

记忆化搜索复杂度分析

  • 时间复杂度：O(n22n) 其中 n 为 nums 的长度。动态规划的时间复杂度 = 状态个数 × 单个状态的计算时间。本题中状态个数等于 O(n2n) ，单个状态的计算时间为 O(n)，因此时间复杂度为 O(n22n) 。
  • 空间复杂度：O(n2n) 。

1:1 翻译成递推

我们可以去掉递归中的「递」，只保留「归」的部分，即自底向上计算。

做法：

  • dfs 改成 f 数组；
  • 递归改成循环（每个参数都对应一层循环）；
  • 递归边界改成 f 数组的初始值。

具体来说，f[i][j] 的含义和状态转移方程 dfs(i,j) 是一样的，即

f(i,j)=∑k∈if[i∖{k}][k]

初始值 f[0][j]=1。（翻译自 dfs(0,j)=1。）

答案为 f[U/{j}][j] 之和。（翻译自 dfs(U∖{j},j) ）

const int MOD = 1e9 + 7;
class Solution {
public:
    int specialPerm(vector<int>& nums) {
        int n = nums.size();
        int s = (1 << n) - 1;
        int ans = 0;
        vector<vector<int>> f(1 << n, vector<int>(n));
        for (int i = 0; i < f.size(); i++) {
            for (int j = 0; j < n; j++) {
                if (i == 0) {
                    f[i][j] = 1;
                }
            }
        }
        for (int i = 1; i < s + 1; i++) {
            for (int j = 0; j < n; j++) {
                for (int k = 0; k < n; k++) {
                    if (((i >> k) & 1) == 1 && (nums[j] % nums[k] == 0 || nums[k] % nums[j] == 0)) {
                        f[i][j] = (f[i][j] + f[i ^ 1 << k][k]) % MOD;
                    }
                }
            }
        }
        for (int i = 0; i < n; i++) {
            ans = (ans + f[s ^ (1 << i)][i]) % MOD;
        }
        return ans;
    }
};

递推复杂度分析

  • 时间复杂度：O(n22n) 其中 n 为 nums 的长度。动态规划的时间复杂度 = 状态个数 × 单个状态的计算时间。本题中状态个数等于 O(n2n) ，单个状态的计算时间为 O(n)，因此时间复杂度为 O(n22n) 。
  • 空间复杂度：O(n2n) 。

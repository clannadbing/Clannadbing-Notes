期望+逆元

4.开关

解题思路

如上图所示，4×4 的网格状

  • 红线：行翻转
  • 蓝线：列翻转

由图可知，一个灯泡最多进行一次行翻转和一次列翻转

第 i 行 j 列灯泡发亮的情况：i 行翻转且 j 列不翻转 + i 行不翻转且 j 列翻转

  • 设 p[i] 为第i 行翻转的概率：p[i]=a[i]/b[i]
  • 设 q[j] 为第j 列翻转的概率: q[j]=c[j]/d[j]

则第 i 行 j 列灯泡发亮的概率: p[i]×(1−q[j])+(1−p[i])×q[j]

题意需要对一个质数取模,除法不能直接取模,可以用费马小定理得逆元，将(除以y)换为(乘以y在%mod下的乘法逆元)，最终结果一样，同时规避了概率是分数的情况

例：对(x/y)取模mod

费马小定理:mod(模数)为质数时，y的逆元为ymod−2 即：xy=x×ymod−2%mod

注: ymod−2 用快速幂计算 (时间复杂度log(mod−2))

  • p[i]=a[i]×b[i]mod−2%mod
  • q[j]=c[j]×d[j]mod−2%mod

答案为每个灯泡概率之和即 ∑i=0∑j=0p[i]⋅(1−q[i])+(1−p[i])⋅q[i]，但是是1e5的数据是不能双重循环的(O(n^2)复杂度)

化简得：

∑i=0∑j=0p[i]⋅(1−q[j])+(1−p[i])⋅q[j]=∑i=0p[i]⋅∑j=0(1−q[j])+∑i=0(1−q[i])∑j=0q[j]=S1×(n−S2)+(n−S1)×S2

S1=p[0]+p[1]+...+p[n]S2=q[0]+q[1]+...+q[n]

实现

#include <iostream>
#include <vector>
using namespace std;
long long mypow(long long x, int n) {
    int MOD = 998244353;
    long long res = 1;
    while (n != 0) {
        if (n & 1) {
            res = res * x % MOD;
        } 
        x = x * x % MOD;
        n = n >> 1;
    }
    return res;
}
int main()
{
    // 请在此输入您的代码
    int MOD = 998244353;
    int n;
    cin >> n;
    vector<int> a(n);
    vector<int> b(n);
    vector<int> c(n);
    vector<int> d(n);
    vector<long long> p(n);
    vector<long long> q(n);
    long long S1 = 0;
    long long S2 = 0;
    for (int i = 0; i < n; i++) {
        cin >> a[i];
    }
    for (int i = 0; i < n; i++) {
        cin >> b[i];
    }
    for (int i = 0; i < n; i++) {
        cin >> c[i];
    }
    for (int i = 0; i < n; i++) {
        cin >> d[i];
    }
    for (int i = 0; i < n; i++) {
        p[i] = a[i] * mypow(b[i], MOD  - 2) % MOD;
        S1 = (S1 + p[i]) % MOD ;
    }
    for (int i = 0; i < n; i++) {
        q[i] = c[i] * mypow(d[i], MOD  - 2) % MOD;
        S2 = (S2 + q[i]) % MOD;
    }
    long long res = (S1 * (n - S2) % MOD + MOD) % MOD + ((n - S1) * S2 % MOD + MOD) % MOD;
    cout << res % MOD << endl;
    return 0;
}
